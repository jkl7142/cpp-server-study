** Critical Section **
// 크리티컬 섹션 객체 생성
CRITICAL_SECTION m_critSec;

// 객체 초기화
InitializeCriticalSectionEx(&m_critSec, 0, 0);

// 객체 삭제
DeleteCriticalSection(&m_critSec);

// 락 Lock()
EnterCriticalSection(&m_critSec);

// 언락 Unlock()
LeaveCriticalSection(&m_critSec);
